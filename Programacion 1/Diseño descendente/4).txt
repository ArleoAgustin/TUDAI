/* Escribir un programa que mientras el usuario cargue desde teclado
 *  un numero entero distinto de 0, imprima por pantalla la suma
	 que se obtiene de invocar un método que calcula la sumatoria de
	 los primeros 200 números naturales (son números enteros en 1 y 200). */

import java.io.BufferedReader;
import java.io.InputStreamReader;
public class entrega {
	public static final int MIN = 1;
	public static final int MAX = 200;
	public static void main(String[] args) {

		sumatoria();
	}
	
//FUNCION CARGAR NUMERO
	public static int cargar_numero () {
		int valor = 0;
		
		BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			
		try {
	
			System.out.println("Ingrese un numero distinto de '0'");
			valor = Integer.valueOf(entrada.readLine());
	
		}
		catch(Exception exc) {
			System.out.println(exc);
		}
		return (valor);
	}
	
//PROCEDIMIENTO SUMATORIA
		public static void sumatoria () {
			int suma;
			suma = 0;
			int valor = cargar_numero();
			if (valor != 0) {
				for (int cont = MIN; cont <= MAX; cont++) {
					suma = suma +cont;
				}
				System.out.println("La sumatoria desde 1 hasta 200 es igual a: " + suma);
			}
		}
		
}
