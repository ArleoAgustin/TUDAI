import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		int valor1, valor2, multiplo;
		boolean resultado;
		multiplo = 2;
		try {
		
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println("Ingrese un numero:");
			valor1 = Integer.valueOf(entrada.readLine());
			System.out.println("Ingrese un numero:");
			valor2 = Integer.valueOf(entrada.readLine());
			resultado = (valor1 > valor2);
			System.out.println("Resultado de (valor1 > valor2): " + resultado);
			valor1 = valor1 %= multiplo;
			valor2 = valor2 %= multiplo;
			System.out.println("Los valores ingresados son multiplo de dos: " + ((valor1 == 0) && (valor2 ==0)));
	}
		catch(Exception exc) {
			System.out.println(exc);
		}
	}
}
