/* Escribir un programa que mientras que el usuario ingrese un número entero
distinto de 0, pida ingresar otro número entero y lo imprima. */

import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		int valor;

		BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
		
		try {
			System.out.println("Ingrese un valor distinto de '0' ");
			valor = Integer.valueOf(entrada.readLine());
			while (valor != 0) {
				System.out.println("");
				System.out.println("Ingrese un valor");
				valor = Integer.valueOf(entrada.readLine());
				System.out.println("");
				System.out.println("El valor ingresado es: " + valor);
				System.out.println("");
				System.out.println("Ingrese un valor, para salir '0' ");
				valor = Integer.valueOf(entrada.readLine());
			}

		}
		catch(Exception exc) {
			System.out.println(exc);
		}
	}
}
