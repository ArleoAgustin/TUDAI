import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		int valor;

		try {
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println("Ingrese un numero");
			valor = Integer.valueOf(entrada.readLine());
			if ((valor % 6 == 0) && (valor % 7 == 0)) {
				System.out.println("El valor es multiplo de 6 y de 7");
			}
			else if ((valor > 30) && (valor % 2 == 0)) {
				System.out.println("El valor es mayor a 30 y es multiplo de 2");
			}
			if (valor % 5 > 30) {
				System.out.println("El cociente de su valor con 5 es mayor a 30");
			}
			else if (valor % 5 < 30) {
				System.out.println("El cociente de su valor con 5 es menor a 30");
			}
		}
		catch(Exception exc) {
			System.out.println(exc);
		}
	}
}
