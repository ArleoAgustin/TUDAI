import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		int valor1, valor2;

		try {
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println("Ingrese un numero");
			valor1 = Integer.valueOf(entrada.readLine());
			System.out.println("Ingrese un numero");
			valor2 = Integer.valueOf(entrada.readLine());
			if (valor1 > valor2) {
				System.out.println("El primero es mayor al segundo: (A)");
			}
			else if ((valor1 % 2 == 0) && ((valor2 % 2 == 0))) {
				System.out.println("Los dos valores son multiplos de 2: (B)");
			}
			else {
				System.out.println("El primer valor es menor o igual al segundo y no son multiplos de 2: (C)");	
			}
			System.out.println("Valor 1: " + valor1);
			System.out.println("Valor 2: " + valor2);
		}
		catch(Exception exc) {
			System.out.println(exc);
		}
	}
}
