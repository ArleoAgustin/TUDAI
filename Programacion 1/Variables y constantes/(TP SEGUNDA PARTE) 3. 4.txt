import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		String TipoFactura, Producto1, Producto2, Nombre;
		double Importe1, Importe2, ImporteTotal;
		int NumFactura;
		
		try {
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println("Ingrese el tipo de factura: ");
			TipoFactura = entrada.readLine();
			System.out.println("Ingrese el numero de factura: ");
			NumFactura = Integer.valueOf(entrada.readLine());
			System.out.println("Ingrese nombre y apellido del cliente: ");
			Nombre = entrada.readLine();
			System.out.println("Ingrese el producto: ");
			Producto1 = entrada.readLine();
			System.out.println("Ingrese el importe del producto: ");
			Importe1 = Double.valueOf(entrada.readLine());
			System.out.println("Ingrese el producto: ");
			Producto2 = entrada.readLine();
			System.out.println("Ingrese el importe del producto: ");
			Importe2 = Double.valueOf(entrada.readLine());
			ImporteTotal = Importe1 + Importe2;
			System.out.println("Nro Factura\t\t\t" + NumFactura);
			System.out.println("Factura\t\t\t\t" + TipoFactura);					
			System.out.println("Nombre y Apellido\t\t" + Nombre);
			System.out.println("Producto\t\t\t\t\t\t\tImporte");
			System.out.println(Producto1 + " \t\t\t\t\t\t\t" + Importe1);
			System.out.println(Producto2 + " \t\t\t\t\t\t\t" + Importe2);
			System.out.println("Importe Total\t\t\t\t\t\t\t"+ ImporteTotal);
		}
			catch (Exception exc) {
					System.out.println(exc);
			}
	}
}
