import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		String Nombre, Ocupacion, Direccion;
		int Edad;
		double Altura;
		try {
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println ("Ingrese su apellido y nombre: ");
			Nombre = entrada.readLine();
			System.out.println("Ingrese su Edad:");
			Edad = Integer.valueOf(entrada.readLine());
			System.out.println("Ingrese su altura: ");
			Altura = Double.valueOf(entrada.readLine());
			System.out.println("Ingrese su ocupacion: ");
			Ocupacion = entrada.readLine();
			System.out.println("Ingrese su direccion: ");
			Direccion = entrada.readLine();
			// En esta parte se van a mostrar las variables por consola
			System.out.println("Apellido y nombre: " + Nombre);
			System.out.println("Edad: " + Edad);
			System.out.println("Altura: " + Altura);
			System.out.println("Ocupacion: " + Ocupacion);
			System.out.println("Direccion: " + Direccion);
			
		}
			catch (Exception exc) {
				System.out.println(exc);
			}
	}

}
