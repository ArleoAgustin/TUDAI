import java.io.BufferedReader;
import java.io.InputStreamReader;
public class Puntodos {
	public static void main(String[] args) {
		char caracter;
		try {
			BufferedReader entrada = new BufferedReader(new InputStreamReader(System.in));
			System.out.println("Ingrese un caracter");
			caracter = entrada.readLine().charAt(0);
			if ((caracter >= '0' ) && (caracter <= '9')) {
				System.out.println("El caracter es un digito: " + caracter);
			}
			else if ((caracter >= 'a') && (caracter <= 'z')) {
					System.out.println("El caracter es una letra minuscula: " + caracter);

							if ((caracter == 'a') || (caracter == 'e') || (caracter == 'i') || (caracter == 'o') || (caracter == 'u')) {
								System.out.println("La letra es una vocal");
							}
							else
								System.out.println("La letra es una consonante");
			}
			else {
				System.out.println("Es otro caracter: " + caracter);
			}
		}
		catch(Exception exc) {
			System.out.println(exc);		
		}
	}
}
