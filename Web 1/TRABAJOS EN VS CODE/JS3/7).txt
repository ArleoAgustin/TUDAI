document.addEventListener('DOMContentLoaded', cargarPagina);
"use strict"
function cargarPagina () {



       document.querySelector(".cant_parrafos").addEventListener("click", function () {

        let imagenes = document.querySelectorAll(".parrafo");
        let contador = 0;

        for(var i = 0; i < imagenes.length; i++) {
                contador++;
        }

        document.querySelector(".resultado1").innerHTML = "LA CANTIDAD DE PARRAFOS ES: " + contador;

        console.log(contador);

       })


       document.querySelector(".cant_items").addEventListener("click", function () {

        let imagenes = document.querySelectorAll("#lista");
        let contador = 0;

        for(var i = 0; i < imagenes.length; i++) {
                contador++;
        }

        document.querySelector(".resultado2").innerHTML = "LA LISTA TIENE " + contador + " ITEMS"

        console.log(contador);

       })


       document.querySelector(".cant_noticias").addEventListener("click", function () {

        let imagenes = document.querySelectorAll(".noticias");
        let contador = 0;

        for(var i = 0; i < imagenes.length; i++) {
                contador++;
        }

        document.querySelector(".resultado3").innerHTML = "LAS SECTIONS DE NOTICIAS SON " + contador;

        console.log(contador);

       })
        
        
}
